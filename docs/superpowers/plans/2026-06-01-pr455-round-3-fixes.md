def _classify_reserved_ip(ip: ipaddress.IPv4Address | ipaddress.IPv6Address) -> str | None:
    """Return a short human-readable category if ``ip`` falls in a
    reserved range that the API must refuse to dial, else None.

    Extracted from the pre-fix inline ladder in ``_validate_cloud_url``
    so Task 2 (B1-NEW pinned-IP propagation) can reuse the same
    classification. MED-1-PARTIAL extends usage from IP-literal-only
    to DNS-resolved hosts.
    """
    if ip.is_loopback:
        return "loopback"
    if ip.is_link_local:
        return "link-local (covers GCE/AWS metadata service)"
    if ip.is_private:
        return "private (RFC1918 / IPv6 ULA)"
    if isinstance(ip, ipaddress.IPv4Address) and ip in ipaddress.ip_network("100.64.0.0/10"):
        return "CGNAT (RFC6598)"
    return None


def _validate_cloud_url(url: str) -> None:
    # ... preserve all pre-existing scheme/host/db/userinfo checks ...

    if os.environ.get("AGNES_ALLOW_RESERVED_CLOUD_URL") == "1":
        return

    host = parsed.host
    if host.lower() == "localhost":
        raise HTTPException(400, detail="cloud_url host is loopback (localhost) — reserved; ...")

    # Build the IP set to classify: either the single literal, or every
    # address `host` resolves to. MED-1-PARTIAL — pre-fix the function
    # returned early on a non-literal hostname, so an attacker-
    # controlled DNS entry pointing at e.g. 169.254.169.254 bypassed.
    try:
        literal = ipaddress.ip_address(host)
        ips_to_check = [literal]
    except ValueError:
        resolved = _resolve_host(host)
        if not resolved:
            # DNS failure: conservative allow — the migrator's TCP
            # connect attempt will fail cleanly downstream, and we
            # don't want to break operators behind broken DNS.
            return
        ips_to_check = []
        for raw in resolved:
            try:
                ips_to_check.append(ipaddress.ip_address(raw))
            except ValueError:
                continue

    for ip in ips_to_check:
        category = _classify_reserved_ip(ip)
        if category is not None:
            raise HTTPException(
                400,
                detail=(
                    f"cloud_url host {host!r} resolves to {ip} which is "
                    f"{category}; reserved"
                ),
            )
```

**Note** on the conservative DNS-failure allow: this preserves a defensible operator-experience baseline (broken DNS doesn't break the API) and lets the migrator surface a clean connection error. Task 2 will tighten the contract further by pinning the IP at queue time.

def _pin_resolved_ip(url: str) -> str | None:
    """Resolve the host in ``url`` to a single IP literal and return
    the URL with that IP substituted. Returns ``None`` if resolution
    fails or the host is already a literal.

    B1-NEW: pre-fix the original hostname was persisted into the job
    JSON; the applier later passed it unresolved to psycopg. A DNS
    rebind between API validation and migrator connect let an attacker-
    controlled host re-resolve to a different IP. Pinning at queue time
    closes the window — the migrator dials the same IP that was
    validated.

    The chosen IP is deterministic (sorted asc) when ``getaddrinfo``
    returns multiple — preserves replay reproducibility.
    """
    from urllib.parse import urlparse, urlunparse
    try:
        parsed = urlparse(url)
    except Exception:
        return None
    host = parsed.hostname
    if not host:
        return None
    try:
        import ipaddress
        ipaddress.ip_address(host)
        return None  # literal already; nothing to pin
    except ValueError:
        pass
    resolved = sorted(_resolve_host(host))
    if not resolved:
        return None
    chosen = resolved[0]
    # Rebuild netloc with the IP. Preserve userinfo, port, scheme,
    # path, query.
    userinfo = ""
    if parsed.username:
        userinfo = parsed.username
        if parsed.password:
            userinfo += f":{parsed.password}"
        userinfo += "@"
    port = f":{parsed.port}" if parsed.port else ""
    new_netloc = f"{userinfo}{chosen}{port}"
    return urlunparse(parsed._replace(netloc=new_netloc))
```

The migrator runs `_check_cancel_before_flip → write_backend_state(target_state)` and `cancel_job` runs `sentinel.touch() → ... → write_backend_state(BackendState(source_backend))` — **neither holds `MigrationLock` during its `write_backend_state` call.** Microsecond window: migrator passes the check, cancel arrives in the gap, cancel reverts to SOURCE, migrator overwrites with TARGET → data on TARGET, instance.yaml on SOURCE (the exact split-brain H1 was meant to prevent). Fix: have the migrator acquire `MigrationLock` around `_check_cancel_before_flip → write_backend_state` AND have `cancel_job` acquire it around `sentinel.touch → write_backend_state`. The lock guarantees atomicity of check-and-flip.

- Modify: `app/api/db_state.py` (`cancel_job` — acquire `MigrationLock` around sentinel + revert)

        BackendState, MigrationLock, MigrationInProgressError, write_backend_state,
    )

    sentinel = _jobs_dir() / f"{job_id}.cancel"
    # H1-PARTIAL: hold MigrationLock across the cancel sentinel write
    # AND the instance.yaml revert. Pre-fix the two operations were
    # separate, leaving a microsecond window where the migrator's
    # _check_cancel_before_flip could pass and then write_backend_state
    # could overwrite the cancel's revert — split-brain end state with
    # data on TARGET but instance.yaml on SOURCE.
    try:
        with MigrationLock():
            sentinel.touch()
            data["status"] = "cancelled"
            # ... existing tmp/replace/chmod block ...
            os.replace(tmp, path)
            os.chmod(path, 0o600)

            source_backend = data["source_backend"]
            revert_url = None if source_backend == "duckdb" else data.get("source_url")
            write_backend_state(BackendState(source_backend), url=revert_url)
    except MigrationInProgressError:
        # The migrator process is currently holding the lock to do its
        # own flip. It will check our sentinel right before the flip;
        # if it's already past the check, the flip will commit and
        # the next cancel call will see status=completed → 409.
        # Briefly retry — the lock is held for at most one flip's
        # worth of file I/O.
        time.sleep(0.5)
        with MigrationLock():
            sentinel.touch()
            data["status"] = "cancelled"
            os.replace(tmp, path)
            os.chmod(path, 0o600)
            source_backend = data["source_backend"]
            revert_url = None if source_backend == "duckdb" else data.get("source_url")
            write_backend_state(BackendState(source_backend), url=revert_url)
